import os
import sys

PASSWORD_FILE = "passwords.txt"
USED_FILE = "used_passwords.txt"

RED = "\033[31m"
BLUE = "\033[34m"
RESET = "\033[0m"

def load_list(filename):
    if not os.path.exists(filename):
        return []
    with open(filename, "r") as f:
        return f.read().splitlines()

def save_list(filename, data):
    with open(filename, "w") as f:
        f.write("\n".join(data) + "\n")

def show_progress(used, remaining):
    k = len(used)
    n = k + len(remaining)
    print(f"ğŸ”¢ {k} of {n} passwords used")
    print(f"â³ Remaining: {n - k} passwords left")

def try_next_password():
    passwords = load_list(PASSWORD_FILE)
    used = load_list(USED_FILE)

    if not passwords:
        print("âœ… All passwords have been used.")
        return

    current = passwords.pop(0)
    used.append(current)

    print(f"\nğŸ” Using password: {RED}{current}{RESET}")
    show_progress(used, passwords)

    # Save updated lists
    save_list(PASSWORD_FILE, passwords)
    save_list(USED_FILE, used)

def show_inventory():
    used = load_list(USED_FILE)
    if not used:
        print("ğŸ“¦ No passwords used yet.")
        return
    print("ğŸ“œ Used password inventory:")
    for pwd in used:
        print(f"{RED}- {pwd}{RESET}")

def reset():
    if os.path.exists(USED_FILE):
        os.remove(USED_FILE)
    print("ğŸ§¹ Inventory reset. Start fresh.")

def main():
    if len(sys.argv) < 2:
        print("Usage: python password_cli.py [use|show|reset]")
        return

    command = sys.argv[1]

    if command == "use":
        try_next_password()
    elif command == "show":
        show_inventory()
    elif command == "reset":
        reset()
    else:
        print("âŒ Unknown command. Use: use | show | reset")

if __name__ == "__main__":
    main()
